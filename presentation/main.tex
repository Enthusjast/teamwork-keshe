    \begin{itemize}

\section[技术原理]{AI图像识别技术原理}

\begin{frame}
    \frametitle{数据采集与预处理}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{图像数据采集}
                首先需要收集大量图片,例如街道场景,动物照片或医疗影像.

                例如自动驾驶公司会收集大量道路图像,用来训练汽车识别车道线,行人和交通标志.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{图像预处理}
                在训练之前,通常需要对图片进行简单处理,例如去除噪声,统一尺寸或调整亮度,使数据更加清晰和一致.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{数据增强}
                通过旋转,裁剪或改变亮度等方式生成新的训练样本.

                例如一张道路图片可以旋转或翻转生成多个样本,从而让模型学到更多不同情况.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{特征提取与分析}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{边缘特征}
                计算机可以检测图像中的边缘,例如物体的轮廓.

                例如在识别人脸时,眼睛,鼻子和嘴巴的轮廓信息非常重要.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{纹理特征}
                纹理反映物体表面的结构,例如草地,布料或动物毛发等.

                通过分析这些纹理特征,计算机可以更好地区分不同物体.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{颜色特征}
                颜色也是识别的重要信息.

                例如交通信号灯和交通标志通常具有明显的颜色特征,因此计算机可以通过颜色快速进行识别.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{模型训练与优化}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{训练数据准备}
                需要对大量图片进行标注,例如标出``猫''``狗''``汽车''等类别.

                这些标注数据就是AI学习的重要材料.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{模型训练}
                通过不断学习这些数据,模型逐渐学会识别不同类型的图像.

                例如研究人员会让计算机反复观察大量图片,并不断调整模型参数.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{模型评估}
                训练完成后,需要用新的图片进行测试.

                如果识别错误较多,就需要继续改进模型或增加训练数据.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{识别与分类算法}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{卷积神经网络(CNN)}
                这是目前最常用的图像识别方法.例如许多手机相机中的``场景识别功能''就是基于这种技术.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{支持向量机(SVM)}
                一种经典的机器学习算法,在早期图像分类任务中被广泛使用.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{目标检测}
                不仅可以识别图像中的物体类别,还可以找出它们在图像中的位置.

                例如自动驾驶系统需要同时识别道路上的行人,自行车和车辆.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\section[伦理问题]{AI图像识别的伦理问题}

\begin{frame}
    \frametitle{数据隐私与安全}
    人脸识别等技术的普及引发了公众对隐私泄露的担忧.
    \begin{itemize}
        \item 个人生物特征信息的过度采集.
        \item 数据库泄露导致的身份冒用风险.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{算法偏见与公平性}
    如果训练数据分布不均,模型可能对特定群体产生偏见.
    \begin{itemize}
        \item \textbf{样本偏差}:对不同肤色或年龄层的人群识别准确率存在差异.
        \item \textbf{决策黑盒}:深度学习模型缺乏可解释性,难以追踪偏见来源.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{对抗攻击风险}
    通过对输入图像添加细微的扰动,可以轻易误导AI模型.
    \begin{itemize}
        \item \textbf{原理}:通过计算梯度,在原图$x$上叠加微小噪声$\varepsilon$,使得$f(x + \varepsilon) \neq f(x)$.
        \item \textbf{威胁}:如在路牌上贴特定贴纸导致自动驾驶系统误判.
    \end{itemize}
\end{frame}
