\section[技术原理]{AI图像识别的技术原理}

\begin{frame}
    \frametitle{数据采集与预处理}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{图像数据采集}
                首先需要收集大量图片,例如街道场景,动物照片或医疗影像.

                例如自动驾驶公司会收集大量道路图像,用来训练汽车识别车道线,行人和交通标志.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{图像预处理}
                在训练之前,通常需要对图片进行简单处理,例如去除噪声,统一尺寸或调整亮度,使数据更加清晰和一致.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{数据增强}
                通过旋转,裁剪或改变亮度等方式生成新的训练样本.

                例如一张道路图片可以旋转或翻转生成多个样本,从而让模型学到更多不同情况.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{层级特征学习}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{浅层特征}
                网络浅层检测基础视觉元素,如边缘,角点和颜色块.

                这些低级特征与具体任务无关,在不同数据集中具有通用性.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{中层特征}
                将浅层边缘组合为纹理,局部形状和图案.

                例如检测车轮的圆形轮廓,眼睛的椭圆形等更复杂的结构.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{高层语义特征}
                网络最深层的神经元对应高级语义概念,如人脸,车辆,文字等.

                层数越深,特征越抽象,与具体分类目标的关联越强.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{前向传播与反向传播}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{前向传播}
                输入图像经过多层卷积,池化等操作,逐层提取特征,最终由全连接层输出分类结果.

                每一层通过非线性激活函数引入表达能力.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{损失函数}
                将网络输出的预测值与真实标签进行比较,计算损失.

                交叉熵损失常用于分类任务:预测越准确,损失值越小.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{反向传播与梯度下降}
                通过链式法则从输出层逐层向前计算每个参数的梯度.

                利用梯度下降更新权重: $w \leftarrow w - \eta \cdot \nabla L$,使损失逐步降低.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{主流网络架构}
    \begin{columns}[t]
        \begin{column}{0.3\textwidth}
            \begin{block}{卷积神经网络(CNN)}
                通过卷积核在图像上滑动提取局部特征,参数共享大幅减少参数量.

                LeNet手写数字识别,AlexNet在ImageNet竞赛中取得突破.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{残差网络(ResNet)}
                引入跳跃连接(skip connection),使梯度可以直接流向浅层.

                解决了深层网络的退化问题,使百层以上的网络训练成为可能.
            \end{block}
        \end{column}
        \begin{column}{0.3\textwidth}
            \begin{block}{Vision Transformer(ViT)}
                将图像分割为固定大小的块,转化为序列输入Transformer.

                在ImageNet上超越同规模CNN,推动了计算机视觉领域的范式转变.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\section[社会安全]{AI图像识别的社会安全}

\begin{frame}
    \frametitle{身份伪装与欺诈}
    \scriptsize

    \begin{columns}[t]
        \begin{column}{0.6\linewidth}
            \textbf{核心问题:深度伪造技术的技术滥用}

            AI生成式技术的爆发让深度伪造(Deepfake)从实验室走向大众.
            借助算法,不法分子可在数秒内制作以假乱真的图像,音频与视频,精准复刻他人面部特征与语音.
            这种技术让身份冒充的门槛大幅降低,无需专业技能即可完成``换脸''与``拟声'',成为实施新型欺诈的核心工具.

            \begin{columns}[t]
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 金融诈骗}
                        \tiny
                        伪造企业高管视频下达转账指令,或冒充亲友进行电信诈骗.
                        利用视听的高度真实性骗取信任,给企业和个人带来直接且巨大的经济损失.
                    \end{block}
                \end{column}
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 名誉损害}
                        \tiny
                        恶意合成虚假不雅或不当言论视频,针对公众人物,企业或普通个人进行抹黑.
                        这种``数字栽赃''破坏力极强,能在短时间内摧毁个人声誉与品牌公信力.
                    \end{block}                    
                \end{column}
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 信任危机}
                        \tiny
                        当``眼见为实''的传统认知被技术彻底颠覆,真实与伪造的边界变得模糊.
                        这不仅增加了信息鉴别的成本,更动摇了社会对影像证据,媒体报道乃至人际互动的基本信任基础.
                    \end{block}
                \end{column}
            \end{columns}
        \end{column}
        \begin{column}{0.4\linewidth}
            \begin{figure}
                \centering
                \includegraphics[width=\linewidth]{figures/fig11.png}
            \end{figure}

            技术原理示意:AI人脸交换技术通过深度学习算法,将目标人物的面部特征无缝映射到源视频中,生成逻辑连贯,视觉逼真的伪造内容.
            这种技术的平民化与易用化,使得身份欺诈的风险呈现指数级上升,成为数字时代个人与机构面临的重大安全挑战.
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{大规模监控与隐私泄露}
    \scriptsize

    \begin{columns}[t]
        \begin{column}{0.6\linewidth}
            \textbf{核心问题:``监控社会''的技术化落地}
            
            无处不在的高清摄像头与具备毫秒级响应的AI识别算法相结合,彻底改变了传统安防的形态.
            这使得对特定个体的大规模,低成本,实时追踪成为常态,物理空间的活动轨迹被实时转化为可存储,可分析的数据资产,
            个人的``不可见性''在技术面前逐渐消失.

            \begin{columns}[t]
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 隐私全面裸奔}
                        \tiny
                        行踪,行为乃至微表情被无孔不入地捕捉,传统的``独处''概念失效.
                        在智能感知下,私人生活的透明度被无限放大,毫无保留.
                    \end{block}
                \end{column}
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 核心数据滥用}
                        \tiny
                        人脸,虹膜等生物特征数据一旦泄露或被滥用,将直接关联个人身份.
                        这种不可更改的数据被窃取后,对人身和财产安全的威胁是不可逆的.
                    \end{block}
                \end{column}
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 自我噤声效应}
                        \tiny
                        当意识到时刻处于``被注视''状态,人们会自发调整行为迎合规训.
                        这种``全景监狱''式的自我审查,将潜移默化地侵蚀社会的思想自由.
                    \end{block}
                \end{column}
            \end{columns}

            \begin{alertblock}{\scriptsize 警示}
                \tiny
                技术的中立性取决于应用边界.在享受智能安防红利的同时,必须为数据采集与使用设定严格的法律红线,
                防止监控权力的无节制扩张,避免技术反噬成为悬在每个人头顶的``达摩克利斯之剑''.
            \end{alertblock}
        \end{column}
        \begin{column}{0.4\linewidth}
            \begin{figure}
                \centering

            \end{figure}

            监控摄像头与智能算法的深度融合,正在将物理世界全面数字化.
            这种无死角的感知能力打破了公共空间与私人领域的传统边界,使个人行为轨迹,生物特征等敏感信息时刻暴露在数据网络中,
            为隐私安全埋下了根本性隐患.
        \end{column}

\begin{frame}
    \frametitle{公共安全与社会稳定威胁}
    \scriptsize

    \begin{columns}[t]
        \begin{column}{0.6\linewidth}
            \textbf{核心问题:AI技术的恶意滥用}

            当AI图像识别与生成技术脱离监管,被不法分子用于非授权目的时,将直接突破传统安防的物理与技术防线.
            这种滥用不仅能规避常规检查,更能通过制造虚假信息引发社会层面的连锁反应,成为威胁公共秩序与国家安全的隐形推手.
            
            \begin{columns}[t]
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 躲避安防检查}
                        \tiny
                        利用3D面具或深度伪造技术复刻面部特征,绕过机场,车站等关键枢纽的人脸识别闸机,
                        使物理安检系统形同虚设,造成潜在的人员与物品流通风险.
                    \end{block}
                \end{column}
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 虚假信息煽动}
                        \tiny
                        通过生成高逼真度的虚假现场影像,在公共网络空间快速传播不实信息.
                        这种``视觉欺骗''极易激化群体对立情绪,引发社会恐慌,破坏正常的社会舆论秩序.
                    \end{block}
                \end{column}
                \begin{column}{0.3\linewidth}
                    \begin{block}{\scriptsize 关键设施渗透}
                        \tiny
                        针对电力,交通等国家关键基础设施的智能视觉监控系统,实施算法干扰或输入欺骗,
                        导致安防系统对异常行为的漏判,直接威胁国家基础设施的运行安全.
                    \end{block}
                \end{column}
            \end{columns}

            \begin{exampleblock}{\scriptsize 本质}
                \tiny
                这类技术滥用本质上是技术能力与安全伦理的失衡.
                它不仅暴露了现有智能安防体系的脆弱性,更对社会治理提出了新的挑战——在享受技术便利的同时,
                必须构建与之匹配的防御机制与法律红线.
            \end{exampleblock}
        \end{column}
